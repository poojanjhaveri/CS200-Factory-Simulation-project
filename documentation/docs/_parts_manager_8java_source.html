637  public boolean accept(File file) {
638  if(file != null) {
639  if(file.isDirectory()||file.toString().equals("pics/parts/no.png")||file.toString().equals("pics/parts/badpart.png"))
640  return false;
641  String extension = getExtension(file);
642  if(extension != null && isSupported(extension))
643  return true;
644  }
645  return false;

648  private String getExtension(File file) {
649  if(file != null) {
650  String filename = file.getName();
651  int dot = filename.lastIndexOf('.');
652  if(dot > 0 && dot < filename.length()-1)
653  return filename.substring(dot+1).toLowerCase();
654  }
655  return null;
656  }

658  private boolean isSupported(String ext) {
659  return ext.equalsIgnoreCase(GIF) || ext.equalsIgnoreCase(PNG) ||
660  ext.equalsIgnoreCase(JPG) || ext.equalsIgnoreCase(BMP) ||
661  ext.equalsIgnoreCase(JPEG);
