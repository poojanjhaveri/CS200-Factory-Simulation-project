119  private static List<Object> immutableListOfType(List<Object> list,
120  Class type, boolean nullOk) {
121  List<Object> l = new ArrayList<Object>(list.size());
122  for (Iterator iter = list.iterator(); iter.hasNext();) {
123  Object item = iter.next();
124  if (item == null) {
125  if (!nullOk) {
126  throw new NullPointerException("item of list is null");
127  }
128  } else if (!type.isInstance(item)) {
129  throw new ClassCastException("item <" + item
130  + "> of list is not an instance of " + type);
131  }
132  l.add(item);
133  }
134  return Collections.unmodifiableList(l);
135  }
