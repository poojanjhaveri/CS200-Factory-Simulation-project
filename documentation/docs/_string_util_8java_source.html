54  return line;
55  }
56  int lineLen = line.length();

58  int oldIdx = 0;
59  int thisIdx;

61  for (int ix = oldIdx; ix < thisIdx; ix++) {
62  sb.append(line.charAt(ix));

64  sb.append(newstr);
65  oldIdx = thisIdx + oldLen;

68  sb.append(line.charAt(ix));

70  return sb.toString();

550  public static String timeIntervalToString(long millis) {
551  StringBuffer sb = new StringBuffer();
552  if (millis < 10 * Constants.SECOND) {
553  sb.append(millis);
554  sb.append("ms");
555  } else {
556  boolean force = false;
557  String stop = null;
558  for (int ix = 0; ix < units.length; ix++) {
559  UD iu = units[ix];
560  long n = millis / iu.millis;
561  if (force || n >= iu.threshold) {
562  millis %= iu.millis;
563  sb.append(n);
564  sb.append(iu.str);
